Check if MapLibre GL JS is supported

Use supported to check for MapLibre GL browser support, and show an alert if the browser does not support MapLibre GL.

<!DOCTYPE html>
<html>
<head>
<meta charset="utf-8" />
<title>Check if MapLibre GL JS is supported</title>
<meta name="viewport" content="initial-scale=1,maximum-scale=1,user-scalable=no" />
<script src="https://unpkg.com/unl-map-js@0.0.1/lib/unl-map-js.js"></script>
<link href="https://unpkg.com/unl-map-js@0.0.1/lib/unl-map-js.css" rel="stylesheet" />
<link
href="https://unpkg.com/maplibre-gl@2.1.9/dist/maplibre-gl.css"
rel="stylesheet"
/>
<style>
	body { margin: 0; padding: 0; }
	#map { position: absolute; top: 0; bottom: 0; width: 100%; }
</style>
</head>
<body>
<div id="map"></div>
 
<script>
if (!UnlSdk.supported()) {
alert("Your browser does not support UNL SDK");
} else {
var map = new UnlSdk.Map({
container: "map",
apiKey: "YOUR-OWN-API-KEY",
vpmId: "YOUR-OWN-VPM-ID",
center: [-74.5, 40],
zoom: 9,
});
}
</script>
 
</body>
</html>
